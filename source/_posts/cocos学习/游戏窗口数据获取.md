           draw () {
        this.node.on(cc.Node.EventType.TOUCH_START, (event: cc.Event.EventTouch)=>{
            var pos = this.node.convertToNodeSpaceAR(event.getLocation());
            var textre = new cc.Texture2D();
            textre.initWithData(this._data, cc.Texture2D.PixelFormat.RGBA8888, this.targetSize.width, this.targetSize.height);
            var _node = new cc.Node();
            _node.addComponent(cc.Sprite).spriteFrame = new cc.SpriteFrame(textre);
            _node.getComponent(cc.Sprite).spriteFrame.setFlipY(true);
            this.node.addChild(_node);
            _node.setPosition(pos);
        })
    }
